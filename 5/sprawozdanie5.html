Cel ćwiczenia¶

Celem ćwiczenia było zapoznanie się z systemami wysokości oraz interpolacją punktów.

Wstęp¶

Wysokość punktu na mapie jest to odległość pomiędzy danym punktem a powierzchnią geoidy. Przeliczenie wysokości z jednego systemu na inny jest możliwe dzięki zastosowaniu interpolacji. Interpolacja jest to metoda wyznaczania wartości nieznanej na podstawie znanych wartości w określonych punktach. W tym przypadku interpolacja jest to wyznaczenie wysokości nieznanego punktu na podstawie znanych wysokości w określonych punktach. W ramach zadania wykorzystałem interpolację do wyznaczenia wysokości danych o przejeździe rowerowym z pliku gpx.

import numpy as np
import gpxpy
import scipy
import gpxpy.gpx
import pandas as pd

def gpx2df(gpx_file_name, new_file_name = 'route_df.csv'):
    with open(gpx_file_name, 'r') as gpx_file:
        gpx = gpxpy.parse(gpx_file)
    route_info = []
    
    for track in gpx.tracks:
        for segment in track.segments:
            for point in segment.points:
                route_info.append({
                    'latitude': point.latitude,
                    'longitude': point.longitude,
                    'elevation': point.elevation,
                    'time':point.time
                })
    
    route_df = pd.DataFrame(route_info)
    
    save_df(route_df, new_file_name)

def blh2xyz(fi, lam, h) -> np.array:
    '''
    Parameters
    ----------
    fi : float
        latitude [rad].
    lam : float
        longitude [rad].
    h : float
        height [m].
    Returns
    -------
    xyz : numpy array
        x, y, z [m].
    '''
    a = 6378137.0
    e2 = 0.00669438002290
    N = a/np.sqrt(1-e2*np.sin(fi)**2)
    x = (N+h)*np.cos(fi)*np.cos(lam)
    y = (N+h)*np.cos(fi)*np.sin(lam)
    z = (N*(1-e2)+h)*np.sin(fi)
    xyz = np.array([x, y, z])
    return xyz

def save_df(route_df, new_file_name= 'route_df.csv'):
    route_df.to_csv(new_file_name, index=False)
    print(f'file saved to {new_file_name}')
def read_route_df(file):
    route_df = pd.read_csv(file)
    return route_df

def interpolate_geoid(model, phi, lam, tr: scipy.spatial.KDTree = None):
    '''
    Parameters
    ----------
    model : numpy array
        geoid model.
    phi : float
        latitude [deg].
    lam : float
        longitude [deg].
    tr : scipy.spatial.KDTree, optional
        KDTree. The default is None.
    Returns
    -------
    h : float
        geoid height [m].
    '''
    if tr is None:
        # https://docs.scipy.org/doc/scipy/reference/generated/scipy.spatial.KDTree.html
        # https://www.youtube.com/watch?v=BK5x7IUTIyU
        # https://en.wikipedia.org/wiki/K-d_tree
        # https://en.wikipedia.org/wiki/Nearest_neighbor_search
        tr = scipy.spatial.KDTree(model[:, :2])
    # find the 4 closest points
    d, idx = tr.query([phi, lam], k=4, workers=-1)
    # calculate weights
    w = 1/d
    # calculate heights
    h = model[idx, 2]
    # calculate weighted average height
    h1 = np.sum(h*w)/np.sum(w)
    return h1

phis = df['latitude'].values
lambdas = df['longitude'].values
dists = []
for i in range(len(phis)-1):
    phi1 = phis[i]
    lam1 = lambdas[i]
    phi2 = phis[i+1]
    lam2 = lambdas[i+1]

    xyz1 = blh2xyz(np.deg2rad(phi1), np.deg2rad(lam1), 0)
    xyz2 = blh2xyz(np.deg2rad(phi2), np.deg2rad(lam2), 0)
    dist = np.linalg.norm(xyz1-xyz2)
    dists.append(dist)

dists.append(dist)
